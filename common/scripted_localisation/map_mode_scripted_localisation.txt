#color surpluss appropriation map mode tooltip values
defined_text = {
	name = get_colored_average_value
	text = {
		trigger = {
			has_country_flag = no_sa_impact
		}
		localization_key = "---"
	}
	text = {
		trigger = {
			check_variable = {
				sa_average_value > 1.5
			}
		}
		localization_key = "[?FROM.controller:sa_average_value|%0B]"
	}
	text = {
		trigger = {
			check_variable = {
				sa_average_value > 0.8
			}
		}
		localization_key = "[?FROM.controller:sa_average_value|%0G]"
	}
	text = {
		trigger = {
			check_variable = {
				sa_average_value > 0.4
			}
		}
		localization_key = "[?FROM.controller:sa_average_value|%0Y]"
	}
	text = {
		localization_key = "[?FROM.controller:sa_average_value|%0R]"
	}
}
