ice_equipment_manpower_collecting = {
	open_tomb = {
		allowed = {
			original_tag = ICE
		}
		days_re_enable = open_tomb_cooldown
        cost = open_tomb_cost
		complete_effect = {
			add_equipment_to_stockpile = {
				type = windigo_equipment
				amount = open_tomb_eq_ammont
			}
		}
		ai_will_do = {
            base = 5000
        }
	}
	destroy_buildings = {
		allowed = {
			original_tag = ICE
		}
		state_target = yes
		target_root_trigger = {
			has_completed_focus = ice_focus_destroy_buildings_for_equipment
		}
		target_array = controlled_states
		on_map_mode = map_and_decisions_view
		days_re_enable = 14
		target_trigger = {
			FROM = {
				OR = {
					check_variable = {
						building_level@arms_factory > 0
					}
					check_variable = {
						building_level@industrial_complex > 0
					}
					check_variable = {
						building_level@agrarian_complex > 0
					}
				}
			}
		}
		days_re_enable = destroy_buildings_cooldown
		cost = 10 #destroy_buildings_cost
		complete_effect = {
			FROM = {
				#"if" isn`t good, but for tooltip - we should leave them (remove text like "remove 0 building")
				if = {
					limit = {
						check_variable = {
							building_level@arms_factory > 0
						}
					}
					remove_building = {
						type = arms_factory
						level = building_level@arms_factory
					}
				}
				if = {
					limit = {
						check_variable = {
							building_level@agrarian_complex > 0
						}
					}
					remove_building = {
						type = agrarian_complex
						level = building_level@agrarian_complex
					}
				}
				if = {
					limit = {
						check_variable = {
							building_level@industrial_complex > 0
						}
					}
					remove_building = {
						type = industrial_complex
						level = building_level@industrial_complex
					}
				}
			}
			set_temp_variable = {
				mil_eq_factor = FROM.building_level@arms_factory
			}
			set_temp_variable = {
				agr_eq_factor = FROM.building_level@agrarian_complex
			}
			set_temp_variable = {
				civ_eq_factor = FROM.building_level@industrial_complex
			}
			multiply_temp_variable = {
				mil_eq_factor = 250
			}
			multiply_temp_variable = {
				agr_eq_factor = 50
			}
			multiply_temp_variable = {
				civ_eq_factor = 100
			}
			add_to_temp_variable = {
				mil_eq_factor = agr_eq_factor
			}
			add_to_temp_variable = {
				mil_eq_factor = civ_eq_factor
			}
			add_equipment_to_stockpile = {
				type = windigo_equipment
				amount = mil_eq_factor
			}
		}
	}
	resurrect_fallen = {
		allowed = {
			original_tag = ICE
		}
		days_re_enable = resurrect_fallen_cooldown
        cost = resurrect_fallen_cost
		complete_effect = {
			set_temp_variable = {
				temp = casualties_k
			}
			subtract_from_temp_variable = {
				temp = resurrect_fallen_prev_casualties_k
			}
			multiply_temp_variable = {
				temp = resurrect_fallen_efficiency
			}
			multiply_temp_variable = {
				temp = 1000
			}
			add_manpower = temp
			set_variable = {
				resurrect_fallen_prev_casualties_k = casualties_k
			}
		}
		ai_will_do = {
            base = 5000
        }
	}
}
