#mobilization law
subject_mobilization_law_0_available = {
	if = {
		limit = {
			has_idea = volunteer_only
		}
		NOT = {
			OR = {
				check_variable = {
					overlord:political_power < mobilization_law_pp_cost_factor
				}
				check_variable = {
					overlord:command_power < mobilization_law_cp_cost_factor
				}
			}
		}
	}
	else = {
		set_temp_variable = {
			pp_cost = mobilization_law_pp_cost_factor
		}
		set_temp_variable = {
			cp_cost = mobilization_law_cp_cost_factor
		}
		if = {
			limit = {
				has_idea = limited_conscription
			}
			multiply_temp_variable = {
				pp_cost = 2
			}
			multiply_temp_variable = {
				cp_cost = 2
			}
		}
		else_if = {
			limit = {
				has_idea = extensive_conscription
			}
			multiply_temp_variable = {
				pp_cost = 3
			}
			multiply_temp_variable = {
				cp_cost = 3
			}
		}
		else_if = {
			limit = {
				has_idea = service_by_requirement
			}
			multiply_temp_variable = {
				pp_cost = 4
			}
			multiply_temp_variable = {
				cp_cost = 4
			}
		}
		else_if = {
			limit = {
				has_idea = all_adults_serve
			}
			multiply_temp_variable = {
				pp_cost = 5
			}
			multiply_temp_variable = {
				cp_cost = 5
			}
		}
		else = {
			multiply_temp_variable = {
				pp_cost = 6
			}
			multiply_temp_variable = {
				cp_cost = 6
			}
		}
		NOT = {
			OR = {
				check_variable = {
					overlord:political_power < pp_cost
				}
				check_variable = {
					overlord:command_power < cp_cost
				}
			}
		}
	}
}
