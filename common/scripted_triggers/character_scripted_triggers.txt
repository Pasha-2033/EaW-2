	OR = {
		has_unit_leader_political_leadership_trait = yes
		has_unit_leader_political_sympathy_trait = yes
	}

has_character_political_loyalty = {
	#PDX can`t use "owner" in localisation stack, setting temp variable
	owner = {
		set_temp_variable = {
			PREV.owner_ideology = current_party_ideology_group
		}
	}
	meta_trigger = {
		text = {
			has_trait = sympathizes_[I]_trait
		}
		I = "[?owner_ideology.GetTokenKey]"
	}
}
