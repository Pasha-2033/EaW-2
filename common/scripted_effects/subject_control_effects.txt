#subject control idea actions
change_subject_mobilization_law_0 = {
	if = {
		limit = {
			has_idea = disarmed_nation
		}
		#pass (for preventing other if checks, optimisation)
	}
	else_if = {
		limit = {
			subject_mobilization_law_0_available = no
		}
		#pass, because this law can`t be set (for preventing other if checks, optimisation)
	}
	else_if = {
		limit = {
			always = no
		}
		#to do trigger that check if idea is applyable for subject and overlord
	}
	else = {
		set_temp_variable = {
			pp_cost = mobilization_law_pp_cost_factor
		}
		set_temp_variable = {
			cp_cost = mobilization_law_cp_cost_factor
		}
		if = {
			limit = {
				has_idea = volunteer_only
			}
			swap_ideas = {
				remove_idea = volunteer_only
				add_idea = disarmed_nation
			}
			multiply_temp_variable = {
				pp_cost = -1
			}
			multiply_temp_variable = {
				cp_cost = -1
			}
		}
		else_if = {
			limit = {
				has_idea = limited_conscription
			}
			swap_ideas = {
				remove_idea = limited_conscription
				add_idea = disarmed_nation
			}
			multiply_temp_variable = {
				pp_cost = -2
				}
			multiply_temp_variable = {
				cp_cost = -2
			}
		}
		else_if = {
			limit = {
				has_idea = extensive_conscription
			}
			swap_ideas = {
				remove_idea = extensive_conscription
				add_idea = disarmed_nation
			}
			multiply_temp_variable = {
				pp_cost = -3
			}
			multiply_temp_variable = {
				cp_cost = -3
			}
		}
		else_if = {
			limit = {
				has_idea = service_by_requirement
			}
			swap_ideas = {
				remove_idea = service_by_requirement
				add_idea = disarmed_nation
			}
			multiply_temp_variable = {
				pp_cost = -4
			}
			multiply_temp_variable = {
				cp_cost = -4
			}
		}
		else_if = {
			limit = {
				has_idea = all_adults_serve
			}
			swap_ideas = {
				remove_idea = all_adults_serve
				add_idea = disarmed_nation
			}
			multiply_temp_variable = {
				pp_cost = -5
			}
			multiply_temp_variable = {
				cp_cost = -5
			}
		}
		else = {
			swap_ideas = {
				remove_idea = scraping_the_barrel
				add_idea = disarmed_nation
			}
			multiply_temp_variable = {
				pp_cost = -6
			}
			multiply_temp_variable = {
				cp_cost = -6
