	for_loop_effect = {
		end = hooftagon^num
		value = i
		add_to_array = {
			PREV.hooftagon = hooftagon^i
		}
	}
	remove_dynamic_modifier = {
		modifier = greate_project_hooftagon_HQ
	}
	remove_dynamic_modifier = {
		modifier = greate_project_hooftagon_FP
	}
	remove_dynamic_modifier = {
		modifier = greate_project_hooftagon_MI
	}
	PREV = {
		add_dynamic_modifier = {
			modifier = greate_project_hooftagon_HQ
		}
		add_dynamic_modifier = {
			modifier = greate_project_hooftagon_FP
		}
		add_dynamic_modifier = {
			modifier = greate_project_hooftagon_MI
		}
	}
	clear_array = hooftagon
}
