increase_science_base_flag = {
	if = {
		limit = {
			has_country_flag = available_proper_science_base
			NOT = {
				has_country_flag = available_outstanding_science_base
			}
		}
		set_country_flag = available_outstanding_science_base
	}

		limit = {
			has_country_flag = available_big_science_base
			NOT = {

			}
		}
		set_country_flag = available_proper_science_base

		limit = {
			NOT = {

			}
		}
		set_country_flag = available_better_science_base

		limit = {
			NOT = {
