# For minor countries where advisors are not defined
# Generate them from a generic template

# Spymasters
# Unable to use random_list for random race portrait, so, mylti_race will have one portrait, like others

every_possible_country = {
	limit = {
		has_dlc = "La Resistance"
		NOT = {
			OR = {
				has_country_flag = no_generic_spymaster
				has_custom_spymaster = yes
				has_allowed_idea_with_traits = {
					idea = head_of_intelligence
					characters = yes
				}
			}
		}
	}
	generate_character = {
		token_base = generic_head_of_intelligence
		name = generic_head_of_intelligence
		advisor = {
			slot = political_advisor
			traits = {
				head_of_intelligence
			}
			visible = {
				can_create_intelligence_agency = yes
			}
			available = {
				has_intelligence_agency = yes
			}
			ai_will_do = {
				factor = 1
				modifier = {
					has_intelligence_agency = no
					factor = 0
				}
			}
		}
	}
	if = {
		limit = {
			is_polar_race = yes
		}
		set_portraits = {
			character = generic_head_of_intelligence
			army = {
				small = GFX_idea_unknown_advisor_bear
			}
		}
	}
	else_if = {
		limit = {
			is_buffalo_race = yes
		}
		set_portraits = {
			character = generic_head_of_intelligence
			army = {
				small = GFX_idea_unknown_advisor_buffalo
			}
		}
	}
	else_if = {
		limit = {
			is_changeling_race = yes
		}
		set_portraits = {
			character = generic_head_of_intelligence
			army = {
				small = GFX_idea_unknown_advisor_bug
			}
		}
	}
	else_if = {
		limit = {
			is_diamond_dog_race = yes
		}
		set_portraits = {
			character = generic_head_of_intelligence
			army = {
				small = GFX_idea_unknown_advisor_dog
			}
		}
	}
	else_if = {
		limit = {
			is_dragon_race = yes
		}
		set_portraits = {
			character = generic_head_of_intelligence
			army = {
				small = GFX_idea_unknown_advisor_dragon
			}
		}
	}
	else_if = {
		limit = {
			is_griffon_race = yes
		}
		set_portraits = {
			character = generic_head_of_intelligence
			army = {
				small = GFX_idea_unknown_advisor_griffon
			}
		}
	}
	else_if = {
		limit = {
			is_minotaur_race = yes
		}
		set_portraits = {
			character = generic_head_of_intelligence
			army = {
				small = GFX_idea_unknown_advisor_minotaur
			}
		}
	}
	else_if = {
		limit = {
			is_penguin_race = yes
		}
		set_portraits = {
			character = generic_head_of_intelligence
			army = {
				small = GFX_idea_unknown_advisor_pingu
			}
		}
	}
	else = {
		set_portraits = {
			character = generic_head_of_intelligence
			army = {
				small = GFX_idea_unknown_advisor
			}
		}
	}
}
