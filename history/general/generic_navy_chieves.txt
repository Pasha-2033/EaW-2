# For minor countries where advisors are not defined
# Generate them from a generic template

# Navy chieves
# Unable to use random_list for random race portrait, so, mylti_race will have one portrait, like others

every_possible_country = {
	limit = {
		NOT = {
			OR = {
				has_country_flag = no_generic_navy_chief_decisive
				has_custom_navy_chieves = yes
				has_allowed_idea_with_traits = {
					idea = navy_chief_decisive_battle_1
					characters = yes
				}
			}
		}
	}
	generate_character = {
		token_base = generic_navy_chief_decisive
		name = generic_navy_chief_decisive
		advisor = {
			slot = navy_chief
			traits = {
				navy_chief_decisive_battle_1
			}
			cost = 100
		}
	}
	if = {
		limit = {
			has_idea = polar_race
		}
		set_portraits = {
			character = generic_navy_chief_decisive
			army = {
				small = GFX_idea_unknown_advisor_bear
			}
		}
	}
	else_if = {
		limit = {
			has_idea = buffalo_race
		}
		set_portraits = {
			character = generic_navy_chief_decisive
			army = {
				small = GFX_idea_unknown_advisor_buffalo
			}
		}
	}
	else_if = {
		limit = {
			has_idea = changeling_race
		}
		set_portraits = {
			character = generic_navy_chief_decisive
			army = {
				small = GFX_idea_unknown_advisor_bug
			}
		}
	}
	else_if = {
		limit = {
			has_idea = diamond_dog_race
		}
		set_portraits = {
			character = generic_navy_chief_decisive
			army = {
				small = GFX_idea_unknown_advisor_dog
			}
		}
	}
	else_if = {
		limit = {
			has_idea = dragon_race
		}
		set_portraits = {
			character = generic_navy_chief_decisive
			army = {
				small = GFX_idea_unknown_advisor_dragon
			}
		}
	}
	else_if = {
		limit = {
			has_idea = griffon_race
		}
		set_portraits = {
			character = generic_navy_chief_decisive
			army = {
				small = GFX_idea_unknown_advisor_griffon
			}
		}
	}
	else_if = {
		limit = {
			has_idea = minotaur_race
		}
		set_portraits = {
			character = generic_navy_chief_decisive
			army = {
				small = GFX_idea_unknown_advisor_minotaur
			}
		}
	}
	else_if = {
		limit = {
			has_idea = penguin_race
		}
		set_portraits = {
			character = generic_navy_chief_decisive
			army = {
				small = GFX_idea_unknown_advisor_pingu
			}
		}
	}
	else = {
		set_portraits = {
			character = generic_navy_chief_decisive
			army = {
				small = GFX_idea_unknown_advisor
			}
		}
	}
}
