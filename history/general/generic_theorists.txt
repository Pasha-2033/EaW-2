# For minor countries where advisors are not defined
# Generate them from a generic template

# Theorists
# Unable to use random_list for random race portrait, so, multi_race will have one portrait, like others

every_possible_country = {
	if = {
		limit = {
			NOT = {
				has_country_flag = no_generic_military_theorist
				has_custom_theorist = yes
				has_allowed_idea_with_traits = {
					idea = military_theorist
					characters = yes
				}
			}
		}
		generate_character = {
			token_base = generic_military_theorist
			name = generic_military_theorist
			advisor = {
				slot = theorist
				traits = {
					military_theorist
				}
				ledger = army
				cost = 100
			}
		}
		set_variable = {
			character_token = token:generic_military_theorist
		}
		generate_portrait = yes
	}
	if = {
		limit = {
			NOT = {
				has_country_flag = no_generic_naval_theorist
				has_custom_theorist = yes
				has_allowed_idea_with_traits = {
					idea = naval_theorist
					characters = yes
				}
			}
		}
		generate_character = {
			token_base = generic_naval_theorist
			name = generic_naval_theorist
			advisor = {
				slot = theorist
				traits = {
					naval_theorist
				}
				ledger = navy
				cost = 100
			}
		}
		set_variable = {
			character_token = token:generic_naval_theorist
		}
		generate_portrait = yes
	}
	if = {
		limit = {
			NOT = {
				has_country_flag = no_generic_air_warfare_theorist
				has_custom_theorist = yes
				has_allowed_idea_with_traits = {
					idea = air_warfare_theorist
					characters = yes
				}
			}
		}
		generate_character = {
			token_base = generic_air_warfare_theorist
			name = generic_air_warfare_theorist
			advisor = {
				slot = theorist
				traits = {
					air_warfare_theorist
				}
				ledger = air
				cost = 100
			}
		}
		set_variable = {
			character_token = token:generic_air_warfare_theorist
		}
		generate_portrait = yes
	}
	#DO NOT DELETE if BELOW#
	#It updates tokens, it MUST be called after latest needed generic character!
	#Otherwise token will be non-zero value, but GetTokenKey and GetTokenLocalizedKey won`t be able to get key!
	#If you have questions - ask leaddev (it`s his code)
	if = {
		limit = {
			always = no
		}
		generate_character = {
			token_base = update_generic_tokens
			name = update_generic_tokens
			advisor = {
				slot = theorist
				traits = {}
				ledger = air
				cost = 0
			}
		}
	}
	########################
}
