# For minor countries where advisors are not defined
# Generate them from a generic template

# Theorists
# Unable to use random_list for random race portrait, so, multi_race will have one portrait, like others

every_possible_country = {
	limit = {
		NOT = {
			OR = {
				has_country_flag = no_generic_military_theorist
				has_custom_theorist = yes
				has_allowed_idea_with_traits = {
					idea = military_theorist
					characters = yes
				}
			}
		}
	}
	generate_character = {
		token_base = generic_military_theorist
		name = generic_military_theorist
		advisor = {
			slot = theorist
			traits = {
				military_theorist
			}
			ledger = army
			cost = 100
		}
	}
	if = {
		limit = {
			has_idea = polar_race
		}
		set_portraits = {
			character = generic_military_theorist
			army = {
				small = GFX_idea_unknown_advisor_bear
			}
		}
	}
	else_if = {
		limit = {
			has_idea = buffalo_race
		}
		set_portraits = {
			character = generic_military_theorist
			army = {
				small = GFX_idea_unknown_advisor_buffalo
			}
		}
	}
	else_if = {
		limit = {
			has_idea = changeling_race
		}
		set_portraits = {
			character = generic_military_theorist
			army = {
				small = GFX_idea_unknown_advisor_bug
			}
		}
	}
	else_if = {
		limit = {
			has_idea = diamond_dog_race
		}
		set_portraits = {
			character = generic_military_theorist
			army = {
				small = GFX_idea_unknown_advisor_dog
			}
		}
	}
	else_if = {
		limit = {
			has_idea = dragon_race
		}
		set_portraits = {
			character = generic_military_theorist
			army = {
				small = GFX_idea_unknown_advisor_dragon
			}
		}
	}
	else_if = {
		limit = {
			has_idea = griffon_race
		}
		set_portraits = {
			character = generic_military_theorist
			army = {
				small = GFX_idea_unknown_advisor_griffon
			}
		}
	}
	else_if = {
		limit = {
			has_idea = minotaur_race
		}
		set_portraits = {
			character = generic_military_theorist
			army = {
				small = GFX_idea_unknown_advisor_minotaur
			}
		}
	}
	else_if = {
		limit = {
			has_idea = penguin_race
		}
		set_portraits = {
			character = generic_military_theorist
			army = {
				small = GFX_idea_unknown_advisor_pingu
			}
		}
	}
	else = {
		set_portraits = {
			character = generic_military_theorist
			army = {
				small = GFX_idea_unknown_advisor
			}
		}
	}
}
